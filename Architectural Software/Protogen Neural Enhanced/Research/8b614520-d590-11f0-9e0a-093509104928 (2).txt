import networkx as nx
import json
import time
from typing import Dict, Any, List, Optional
from pathlib import Path

# Assuming SuperQuantumToken is defined in a sibling file
from sqt import SuperQuantumToken
from progress_tracker import ProgressTracker 

class OntologyGraph:
    """
    Manages the interconnected network of SuperQuantumTokens (SQTs),
    forming the evolving ontology and knowledge representation.
    This is the system's semantic knowledge graph.
    """
    def __init__(self, storage_path: Path, progress_tracker=None):
        self.storage_path = storage_path
        self.graph_file = self.storage_path / "ontology_graph.json"
        self.sqt_cache_file = self.storage_path / "sqt_cache.json" # To store SQT objects

        self.graph = nx.DiGraph() # Directed graph for conceptual links
        self.sqt_register: Dict[str, SuperQuantumToken] = {} # Hash -> SQT Object mapping

        self.progress = progress_tracker or ProgressTracker(verbose=False) # Ensure progress tracker is initialized
        self._load_state()

    def _load_state(self):
        """Loads the graph and SQT register from storage."""
        self.storage_path.mkdir(parents=True, exist_ok=True)
        
        if self.graph_file.exists():
            try:
                # networkx can't directly load custom objects, so we store node data
                with open(self.graph_file, 'r', encoding='utf-8') as f:
                    graph_data = json.load(f)
                self.graph = nx.node_link_graph(graph_data)
            except (json.JSONDecodeError, Exception) as e:
                print(f"Warning: Could not load graph data, starting fresh. Error: {e}")
                self.graph = nx.DiGraph()

        if self.sqt_cache_file.exists():
            try:
                with open(self.sqt_cache_file, 'r', encoding='utf-8') as f:
                    sqt_data = json.load(f)
                for s_hash, s_dict in sqt_data.items():
                    # Reconstruct SQT objects with new ethical_valence parameter
                    self.sqt_register[s_hash] = SuperQuantumToken(
                        concept_id=s_dict['concept_id'],
                        conceptual_type=s_dict['conceptual_type'],
                        description=s_dict['description'],
                        source_data_hash=s_dict.get('source_data_hash'),
                        initial_embedding=s_dict.get('initial_embedding'),
                        ethical_valence=s_dict.get('ethical_valence') # Load new attribute
                    )
                    # Manually set hash and timestamp if not passed in constructor for fidelity
                    self.sqt_register[s_hash].hash = s_hash
                    self.sqt_register[s_hash].creation_timestamp = s_dict['creation_timestamp']

            except (json.JSONDecodeError, Exception) as e:
                print(f"Warning: Could not load SQT cache, starting fresh. Error: {e}")
                self.sqt_register = {}
                
    def _save_state(self):
        """Saves the graph and SQT register to storage."""
        with open(self.graph_file, 'w', encoding='utf-8') as f:
            json.dump(nx.node_link_data(self.graph), f, indent=4)
        
        sqt_data_to_save = {s_hash: sqt.to_dict() for s_hash, sqt in self.sqt_register.items()}
        with open(self.sqt_cache_file, 'w', encoding='utf-8') as f:
            json.dump(sqt_data_to_save, f, indent=4)

    def add_sqt(self, sqt: SuperQuantumToken):
        """Adds an SQT to the ontology graph."""
        if sqt.hash not in self.sqt_register:
            self.sqt_register[sqt.hash] = sqt
            # Store SQT data including ethical_valence as node attributes
            self.graph.add_node(sqt.hash, **sqt.to_dict()) 
            self.progress.log_action(f"Added SQT: {sqt.concept_id} (Valence: {sqt.ethical_valence})")
            self._save_state()
        else:
            # If SQT exists, consider updating its attributes, especially if ethical_valence changes
            existing_sqt = self.sqt_register[sqt.hash]
            if existing_sqt.ethical_valence != sqt.ethical_valence:
                existing_sqt.ethical_valence = sqt.ethical_valence
                self.graph.nodes[sqt.hash]['ethical_valence'] = sqt.ethical_valence # Update node attribute
                self.progress.log_action(f"Updated SQT {sqt.concept_id} ethical valence to {sqt.ethical_valence}")
                self._save_state()
            # print(f"SQT {sqt.concept_id} already exists.") # Keep this for debugging if needed

    def add_conceptual_link(self, source_sqt_hash: str, target_sqt_hash: str, weight: float = 1.0, ethical_valence_contribution: Optional[float] = None):
        """
        Adds or strengthens a directed conceptual link between two SQTs.
        Weight represents the strength/frequency of the perceived relationship.
        ethical_valence_contribution: New field to denote ethical impact of this specific link.
        """
        if source_sqt_hash not in self.sqt_register or target_sqt_hash not in self.sqt_register:
            raise ValueError("Both source and target SQTs must exist in the register.")
        
        if self.graph.has_edge(source_sqt_hash, target_sqt_hash):
            self.graph[source_sqt_hash][target_sqt_hash]['weight'] += weight
            # Update ethical_valence_contribution if provided, or average/combine
            if ethical_valence_contribution is not None:
                current_contrib = self.graph[source_sqt_hash][target_sqt_hash].get('ethical_valence_contribution', 0.0)
                self.graph[source_sqt_hash][target_sqt_hash]['ethical_valence_contribution'] = (current_contrib + ethical_valence_contribution) / 2 # Simple averaging
            self.progress.log_action(f"Strengthened link from {self.sqt_register[source_sqt_hash].concept_id} to {self.sqt_register[target_sqt_hash].concept_id}. New weight: {self.graph[source_sqt_hash][target_sqt_hash]['weight']}")
        else:
            self.graph.add_edge(source_sqt_hash, target_sqt_hash, weight=weight, ethical_valence_contribution=ethical_valence_contribution)
            self.progress.log_action(f"Created link from {self.sqt_register[source_sqt_hash].concept_id} to {self.sqt_register[target_sqt_hash].concept_id} with weight: {weight}, ethical contribution: {ethical_valence_contribution}")
        self._save_state()

    def get_sqt_by_hash(self, sqt_hash: str) -> Optional[SuperQuantumToken]:
        """Retrieves an SQT object by its hash."""
        return self.sqt_register.get(sqt_hash)

    def get_neighbors(self, sqt_hash: str) -> List[Dict[str, Any]]:
        """Returns a list of connected SQTs and their link weights."""
        neighbors = []
        if sqt_hash in self.graph:
            for neighbor_hash, data in self.graph[sqt_hash].items():
                neighbor_sqt = self.get_sqt_by_hash(neighbor_hash)
                if neighbor_sqt:
                    neighbors.append({
                        "sqt": neighbor_sqt,
                        "weight": data.get('weight', 1.0),
                        "ethical_valence_contribution": data.get('ethical_valence_contribution') # Include new attribute
                    })
        return neighbors

    def calculate_eigenvector_centrality(self, graph=None) -> Dict[str, float]:
        """
        Calculates the eigenvector centrality for each node (SQT) in the graph.
        This helps identify Axiomatic Anchors. Can optionally work on a provided graph.
        """
        graph_to_use = graph if graph is not None else self.graph
        if not graph_to_use.nodes():
            return {}
        try:
            # Using weight as the attribute for centrality calculation
            centrality = nx.eigenvector_centrality(graph_to_use, weight='weight', max_iter=1000, tol=1e-06)
            return centrality
        except nx.PowerIterationFailedConvergence:
            # print("Warning: Eigenvector centrality failed to converge. Returning degree centrality as fallback.")
            return nx.degree_centrality(graph_to_use)

    def identify_knowledge_gaps(self, min_connectivity: int = 2, max_depth: int = 3, core_sqt_hashes: Optional[List[str]] = None) -> List[Dict[str, Any]]:
        """
        Identifies potential knowledge gaps or areas of low connectivity within the ontology.
        Can be guided by 'core_sqt_hashes' (e.g., purpose-defining SQTs) to find gaps related to Protogen's purpose.
        Returns a list of identified gaps, e.g., isolated SQTs, weakly connected clusters, or areas
        lacking connections to core concepts.
        """
        self.progress.start_operation("Identifying Knowledge Gaps")
        gaps = []
        
        # 1. Identify isolated or weakly connected nodes
        isolated_sqts = [self.get_sqt_by_hash(node) for node in self.graph.nodes() if self.graph.degree(node) < min_connectivity]
        if isolated_sqts:
            gaps.append({"type": "Weakly Connected Concepts", "details": [s.concept_id for s in isolated_sqts if s]})
            self.progress.log_action(f"Found {len(isolated_sqts)} weakly connected concepts.")

        # 2. Identify missing links to core SQTs (if provided)
        if core_sqt_hashes:
            core_sqt_set = set(core_sqt_hashes)
            for node_hash in self.graph.nodes():
                if node_hash not in core_sqt_set:
                    # Check if node is well-connected to any core SQT
                    connected_to_core = False
                    for core_hash in core_sqt_set:
                        if nx.has_path(self.graph, node_hash, core_hash) or nx.has_path(self.graph, core_hash, node_hash):
                            # Also check path length
                            if nx.shortest_path_length(self.graph, node_hash, core_hash) <= max_depth or nx.shortest_path_length(self.graph, core_hash, node_hash) <= max_depth:
                                connected_to_core = True
                                break
                    if not connected_to_core:
                        gaps.append({"type": "Disconnected from Core", "concept_id": self.get_sqt_by_hash(node_hash).concept_id if self.get_sqt_by_hash(node_hash) else "Unknown", "reason": f"No direct path within {max_depth} to core concepts."})
                        self.progress.log_action(f"Found concept disconnected from core: {self.get_sqt_by_hash(node_hash).concept_id}")

        # 3. Identify potential contradictions (conceptual - requires reasoning engine)
        # This is more complex and would ideally involve the ReasoningEngine to detect logical inconsistencies
        # For now, it's a placeholder. Example: SQTs with highly opposing ethical valences linked directly.
        contradictory_links = []
        for u, v, data in self.graph.edges(data=True):
            if 'ethical_valence_contribution' in data and data['ethical_valence_contribution'] < -0.5: # Example threshold for contradiction
                sqt_u = self.get_sqt_by_hash(u)
                sqt_v = self.get_sqt_by_hash(v)
                if sqt_u and sqt_v and sqt_u.ethical_valence is not None and sqt_v.ethical_valence is not None and sqt_u.ethical_valence * sqt_v.ethical_valence < 0 and data['ethical_valence_contribution'] < 0: # Concepts with opposing valences linked negatively
                     contradictory_links.append(f"{sqt_u.concept_id} <-> {sqt_v.concept_id} (valence: {data['ethical_valence_contribution']:.2f})")
        if contradictory_links:
            gaps.append({"type": "Potential Ethical Contradictions", "details": contradictory_links})
            self.progress.log_action(f"Found {len(contradictory_links)} potential ethical contradictions.")

        self.progress.end_operation(success=True)
        return gaps

    def get_graph_summary(self) -> Dict[str, Any]:
        """Provides a summary of the graph's current state."""
        return {
            "num_sqts": self.graph.number_of_nodes(),
            "num_links": self.graph.number_of_edges(),
            "is_directed": self.graph.is_directed(),
            "last_saved": time.time_ns()
        }

                     ethical_valence: Optional[float] = None):

        def _calculate_hash(self):
            import hashlib
            data_string = f"{self.concept_id}-{self.conceptual_type}-{self.description}-{self.creation_timestamp}"
            return hashlib.sha256(data_string.encode('utf-8')).hexdigest()

        def to_dict(self):
            return {
                "concept_id": self.concept_id,
                "conceptual_type": self.conceptual_type,
                "description": self.description,
                "source_data_hash": self.source_data_hash,
